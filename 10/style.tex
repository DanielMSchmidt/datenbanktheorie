\def\ukapitel#1{

\vspace{1.5cm}

\addcontentsline{toc}{subsection}{\quad\theukapitel. #1}


\bigskip

}

\setcounter{ukapitel}{0}\setcounter{uukapitel}{1}

\bigskip

}

\vspace{1cm}

}
